\begin{itemize}
\item \keyword{Eventi indipendenti}: joint probability equals product of probability
\[\prob{(A\cap B)}=\prob{(A)}\prob{(B)}\Leftrightarrow \prob{(A)}=\frac{\prob{(A\cap B)}}{\prob{(B)}}=\prob{A|B}\]
\item \keyword{teorema probabilit\'a totale}: $A\cap B=\emptyset$, $\prob{(A\cap B)}=\prob{(A)}+\prob{(B)}$
\item \keyword{disuguaglianza di Chebychev}: X ha media $\mu$ e var $\sigma^2$ e $\lambda>0$:
\[\prob{(|X-\mu|<\lambda\sigma)}\geq1-\frac{1}{\lambda^2}\]

\end{itemize}
